Cleanup Functions Report (Detailed)

Each section explains why cleanup is required, what can break if skipped, and includes a detailed TypeScript snapshot using a shared cleanup registry.

Shared Pattern: Cleanup Registry

Use one registry per root owner (screen/component/feature) and execute all cleanup functions on teardown.

type CleanFn = () => void;
const cleanUpMap = new WeakMap<HTMLElement, CleanFn[]>();

export function addCleanup(root: HTMLElement, fn: CleanFn): void {
  const queue = cleanUpMap.get(root) ?? [];
  queue.push(fn);
  cleanUpMap.set(root, queue);
}

export function runAllCleanups(root: HTMLElement): void {
  const queue = cleanUpMap.get(root) ?? [];
  for (const fn of queue) fn();
  queue.length = 0;
}

1. setInterval

  • Leak risk: interval keeps running after UI is gone.
  • Symptom: duplicated logs, duplicated network calls, CPU usage.
let tick = 0;
const intervalId = window.setInterval(() => {
  tick += 1;
  counterEl.textContent = String(tick);
}, 1000);

addCleanup(root, () => {
  window.clearInterval(intervalId);
});

2. setTimeout

  • Timeout may fire after unmount and mutate stale DOM/state.
const timeoutId = window.setTimeout(() => {
  toastEl.classList.add("visible");
}, 300);

addCleanup(root, () => {
  window.clearTimeout(timeoutId);
});

3. Element Event Binding

  • Always keep a stable handler reference.
  • Inline functions cannot be removed later.
const onStartClick = () => startWorkflow();
startBtn.addEventListener("click", onStartClick);

addCleanup(root, () => {
  startBtn.removeEventListener("click", onStartClick);
});

4. Window Event Binding

  • Global listeners outlive local UI if not removed.
const onResize = () => {
  const compact = window.innerWidth < 900;
  layoutEl.dataset.compact = String(compact);
};
window.addEventListener("resize", onResize);

addCleanup(root, () => {
  window.removeEventListener("resize", onResize);
});

5. Document Delegated Listener

  • Useful for dynamic nodes; must still be cleaned.
const onDocClick = (ev: MouseEvent) => {
  const target = ev.target as HTMLElement | null;
  if (target?.closest("[data-action='close']")) panelEl.classList.add("hidden");
};
document.addEventListener("click", onDocClick);

addCleanup(root, () => {
  document.removeEventListener("click", onDocClick);
});

6. MutationObserver

  • Observer callbacks continue unless disconnected.
const mo = new MutationObserver((records) => {
  for (const record of records) {
    console.log("changed nodes:", record.addedNodes.length);
  }
});
mo.observe(listEl, { childList: true, subtree: true });

addCleanup(root, () => mo.disconnect());

7. ResizeObserver

  • Can fire frequently; disconnect on teardown.
const ro = new ResizeObserver((entries) => {
  const width = entries[0]?.contentRect.width ?? 0;
  titleEl.textContent = `Width: ${width.toFixed(0)}px`;
});
ro.observe(panelEl);

addCleanup(root, () => ro.disconnect());

8. IntersectionObserver

  • Lazy-loading/analytics callbacks should stop when feature is removed.
const io = new IntersectionObserver((entries) => {
  for (const entry of entries) {
    if (entry.isIntersecting) loadPreview();
  }
});
io.observe(imageEl);

addCleanup(root, () => io.disconnect());

9. requestAnimationFrame

  • Pending frame callback should be canceled if animation owner disappears.
let rafId = 0;
const animate = () => {
  draw();
  rafId = window.requestAnimationFrame(animate);
};
rafId = window.requestAnimationFrame(animate);

addCleanup(root, () => window.cancelAnimationFrame(rafId));

10. requestIdleCallback

  • Idle work can run late; cancel if no longer relevant.
const idleId = window.requestIdleCallback(() => {
  buildSearchIndex();
});

addCleanup(root, () => {
  window.cancelIdleCallback(idleId);
});

11. AbortController (generic async)

  • Stops long async operations and prevents post-unmount side effects.
const controller = new AbortController();
runLongTask({ signal: controller.signal })
  .catch((err) => {
    if (err.name !== "AbortError") throw err;
  });

addCleanup(root, () => controller.abort());

12. fetch with AbortController

  • Abort in-flight request to avoid updating detached UI.
const controller = new AbortController();
fetch("/api/registers", { signal: controller.signal })
  .then((res) => res.json())
  .then((rows) => renderRows(rows))
  .catch((err) => {
    if (err.name !== "AbortError") console.error(err);
  });

addCleanup(root, () => controller.abort());

13. WebSocket

  • Remove handlers and close socket explicitly.
const ws = new WebSocket("wss://example.com/realtime");
const onMessage = (ev: MessageEvent) => {
  appendMessage(JSON.parse(String(ev.data)));
};
ws.addEventListener("message", onMessage);

addCleanup(root, () => {
  ws.removeEventListener("message", onMessage);
  ws.close();
});

14. EventSource (SSE)

  • SSE remains open until you call close().
const sse = new EventSource("/stream/updates");
const onUpdate = (ev: MessageEvent) => {
  updateBadge(ev.data);
};
sse.addEventListener("message", onUpdate);

addCleanup(root, () => {
  sse.removeEventListener("message", onUpdate);
  sse.close();
});

15. BroadcastChannel

  • Cross-tab messages should stop after feature teardown.
const channel = new BroadcastChannel("registers-sync");
const onChannelMessage = (ev: MessageEvent) => {
  mergeState(ev.data);
};
channel.addEventListener("message", onChannelMessage);

addCleanup(root, () => {
  channel.removeEventListener("message", onChannelMessage);
  channel.close();
});

16. MediaQueryList listener

  • Responsive/theme watchers are global and persistent if not removed.
const mql = window.matchMedia("(prefers-reduced-motion: reduce)");
const onMqlChange = (ev: MediaQueryListEvent) => {
  document.body.dataset.reduceMotion = String(ev.matches);
};
mql.addEventListener("change", onMqlChange);

addCleanup(root, () => mql.removeEventListener("change", onMqlChange));

17. Geolocation watchPosition

  • Always clear watch IDs to stop tracking callbacks.
const watchId = navigator.geolocation.watchPosition(
  (pos) => {
    latEl.textContent = pos.coords.latitude.toFixed(4);
    lngEl.textContent = pos.coords.longitude.toFixed(4);
  },
  (err) => console.error(err)
);

addCleanup(root, () => navigator.geolocation.clearWatch(watchId));

18. Router/Store subscription

  • If API returns unsubscribe, register it as cleanup immediately.
const unsubscribe = store.subscribe((nextState) => {
  renderState(nextState);
});

addCleanup(root, () => unsubscribe());

19. Third-party instances

  • Libraries usually expose destroy/dispose.
const editor = createEditor(editorEl, { language: "ts" });
editor.setValue("// hello");

addCleanup(root, () => {
  editor.destroy();
});

20. Dynamic style/script/node injection

  • Remove injected assets that belong to this feature.
const styleEl = document.createElement("style");
styleEl.textContent = ".registers-outline { outline: 2px solid #0ea5e9; }";
document.head.appendChild(styleEl);

const markerEl = document.createElement("div");
markerEl.className = "registers-outline";
document.body.appendChild(markerEl);

addCleanup(root, () => {
  styleEl.remove();
  markerEl.remove();
});
